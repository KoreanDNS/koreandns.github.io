```c++
// 처음에 익은 토마토를 모두 큐에 담아주는게 핵심이다.

#include <bits/stdc++.h>
using namespace std;

const int MAX = 1000 + 1;

int board[MAX][MAX];
int vis[MAX][MAX];

int main(void)
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n, m;
	cin >> m >> n;

	queue<pair<int, int>> q;
	for (int yIndex = 0; yIndex < n; ++yIndex)
	{
		for (int xIndex = 0; xIndex < m; ++xIndex)
		{
			cin >> board[yIndex][xIndex];
			vis[yIndex][xIndex] = -1;
			if (1 == board[yIndex][xIndex])
			{
				q.push({yIndex, xIndex});
				vis[yIndex][xIndex] = 0;
			}
		}
	}

	int dy[4]{ -1, 1, 0, 0 };
	int dx[4]{ 0, 0, -1, 1 };

	int ans = 0;
	while (false == q.empty())
	{
		auto cur = q.front();
		q.pop();

		for (int dir = 0; dir < 4; ++dir)
		{
			int ny = cur.first + dy[dir];
			int nx = cur.second + dx[dir];

			if (ny < 0 or MAX <= ny or nx < 0 or MAX <= nx)
			{
				continue;
			}

			if (0 == board[ny][nx] and -1 == vis[ny][nx])
			{
				vis[ny][nx] = vis[cur.first][cur.second] + 1;
				ans = vis[ny][nx];
				q.push({ ny, nx });
			}
		}
	}

	for (int yIndex = 0; yIndex < n; ++yIndex)
	{
		for (int xIndex = 0; xIndex < m; ++xIndex)
		{
			if (0 == board[yIndex][xIndex] and -1 == vis[yIndex][xIndex])
			{
				ans = -1;
				goto JUMP;
			}
		}
	}

JUMP:
	cout << ans;

	return 0;
}
```
