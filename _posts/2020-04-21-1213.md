```c++
#include <bits/stdc++.h>
using namespace std;

int main(void)
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	string msg;
	cin >> msg;

	//정렬 한 다음에 풀어야하는 문제
	sort(msg.begin(), msg.end());

	// 알파벳 A~Z 26개
	const int LENGTH = 26;
	int arrD[LENGTH] = { 0, };
	for (auto ch : msg)
	{
		arrD[ch - 'A']++;
	}

	// 펠린드롬 할지 말지 여부 판단
	int oddNumberCount = 0;
	int evenNumberCount = 0;
	int oddNumberIndex = 0;
	for (int index = 0; index < LENGTH; ++index)
	{
		int val = arrD[index];
		arrD[index] = val / 2;
		evenNumberCount += (val / 2);
		oddNumberCount += val % 2;

		//홀수 개가 2개 이상이라면 검사 할 필요가 없음
		if (1 < oddNumberCount)
		{
			cout << "I'm Sorry Hansoo";
			return 0;
		}

		if (0 != val % 2)
		{
			//홀수 캐릭터가 뭔지 기억하기 위해서
			oddNumberIndex = index;
		}
	}

	//여기서 부터는 무조건 팰린드롬 성공
	int vLen = (2 * evenNumberCount) + oddNumberCount;
	vector<char> v(vLen);

	//짝수 양옆에 추가
	int msgCount = 0;
	for (int index = 0; index < LENGTH; ++index)
	{
		int val = arrD[index];
		for (int msgIndex = 0; msgIndex < val; ++msgIndex)
		{
			v[msgCount] = 'A' + index;
			v[vLen - 1 - msgCount] = 'A' + index;
			msgCount++;
		}
	}

	//만약 홀수가 있다면 홀수 추가
	if (oddNumberCount)
	{
		v[vLen / 2] = 'A' + oddNumberIndex;
	}

	//결과 출력
	for (auto pr : v)
	{
		cout << pr;
	}

	return 0;
}
```
