```c++
#include <string>
#include <vector>
#include <queue>

using namespace std;

const int MAX = 200 + 1;
bool vis[MAX];

int solution(int n, vector<vector<int>> computers)
{
	int answer = 0;

	queue<int> q;
	for (int index = 0; index < n; ++index)
	{
		if (vis[index])
		{
			continue;
		}

		++answer;

		q.push(index);
		vis[index] = true;

		while (false == q.empty())
		{
			auto cur = q.front();
			q.pop();

			for (int e = 0; e < computers[cur].size(); ++e)
			{
				if (0 == computers[cur][e] or vis[e])
				{
					continue;
				}

				q.push(e);
				vis[e] = true;
			}
		}
	}

	return answer;
}
```
