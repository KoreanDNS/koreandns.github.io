오.. 생각보다 재밌는 문제였다. 그리고 다른 사람 풀이 방법도 다양하구나



```c++
#include <string>
#include <vector>
#include <queue>

using namespace std;

int solution(vector<int> numbers, int target) {
    int answer = 0;
    
    queue<pair<int, int>> q;

	q.push({ -numbers[0], 0 });
	q.push({ numbers[0], 0 });

	while (false == q.empty())
	{
		auto cur = q.front();
		q.pop();

		int data = cur.first;
		int index = cur.second;
		int nextIndex = index + 1;
		if (numbers.size() == index + 1)
		{
			if (target == cur.first)
			{
				++answer;
			}
			continue;
		}

		q.push({ data + numbers[nextIndex], nextIndex });
		q.push({ data - numbers[nextIndex], nextIndex });
	}
    
    return answer;
}
```
