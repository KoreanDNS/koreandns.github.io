링크 : [https://www.acmicpc.net/problem/13335](https://www.acmicpc.net/problem/13335)



풀이

1. 트럭 여러 대가 일 차선 다리를 정해진 순으로 건너려 할때 최소 몇초가 걸리는지 구현하면 된다.
2. 큐를 사용하여 트럭의 무게와, 다리의 길이를 넣어주고 시뮬레이션 해 주었다.



```c++
#include <bits/stdc++.h>
using namespace std;

const int MAX = 1001;
int board[MAX];

int main()
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n, w, l;
	cin >> n >> w >> l; // w : 다리의 길이 l : 다리의 최대 하중
	
	for (int idx=0; idx<n; ++idx)
	{
		cin >> board[idx];
	}

	int ans = 0;
	queue<pair<int, int>> q;
	for (int i=0; i<n;)
	{
		int what = board[i];
		if (what <= l)
		{
			l -= what;
			q.push({ what, w });
			++i;
		}
		
		const int QSIZE = q.size();
		for (int j=0; j<QSIZE; ++j)
		{
			auto cur = q.front();
			q.pop();
			if (0 == --cur.second)
			{
				l += cur.first;
				continue;
			}
			q.push(cur);
		}
		++ans;
	}

	int QSIZE;
	while (QSIZE = q.size())
	{
		for (int j = 0; j<QSIZE; ++j)
		{
			auto cur = q.front();
			q.pop();
			if (0 == --cur.second)
			{
				
				l += cur.first;
				continue;
			}
			q.push(cur);
		}
		++ans;
	}

	cout << ans+1;

	return 0;
}
```
