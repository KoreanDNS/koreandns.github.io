링크 : [https://www.acmicpc.net/problem/1935](https://www.acmicpc.net/problem/1935)



```c++
#include <bits/stdc++.h>
using namespace std;

int main(void)
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int t;
	cin >> t;

	string msg;
	cin >> msg;

	double data[27];
	for (int index = 0; index < t; ++index)
	{
		double n;
		cin >> n;
		data[index] = n;
	}

	stack<double> s;
	for (auto cur : msg)
	{
		double num1, num2;
		if (cur < 'A' or 'Z' < cur)
		{
			num1 = s.top();
			s.pop();
			num2 = s.top();
			s.pop();
		}

		switch (cur)
		{
		case '*':
			s.push(num2 * num1);
			break;
		case '+':
			s.push(num2 + num1);
			break;
		case '/':
			s.push(num2 / num1);
			break;
		case '-':
			s.push(num2 - num1);
			break;
		default:
			s.push(data[cur - 'A']);
			break;
		}
	}

	cout << fixed << setprecision(2) << s.top();
	return 0;
}
```
