링크 : [https://www.acmicpc.net/problem/2206](https://www.acmicpc.net/problem/2206)



이 문제 나름 푸는데 재밌었다고 생각한다..

bfs 문제 + 벽인지 아닌지 상태를 저장하는 공간 이용한다면 풀 수 있는 문제



```c++
#include <bits/stdc++.h>
using namespace std;

const int MAX = 1000 + 1;

char board[MAX][MAX];
int vis[MAX][MAX][2];

int main(void)
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n, m;
	cin >> n >> m;

	for (int yIndex = 0; yIndex < n; ++yIndex)
	{
		for (int xIndex = 0; xIndex < m; ++xIndex)
		{
			cin >> board[yIndex][xIndex];
		}
	}

	queue<tuple<int, int, int>> q;
	q.push({ 0, 0, 1 });
	vis[0][0][0] = 1;
	vis[0][0][1] = 1;

	int dy[4]{ -1, 1, 0, 0 };
	int dx[4]{ 0, 0, -1, 1 };

	while (false == q.empty())
	{
		auto cur = q.front();
		q.pop();

		int cy, cx, cz;
		tie(cy, cx, cz) = cur;

		if (n - 1 == cy and m - 1 == cx)
		{
			cout << vis[cy][cx][cz];
			return 0;
		}

		for (int dir = 0; dir < 4; ++dir)
		{
			int ny = cy + dy[dir];
			int nx = cx + dx[dir];
			int nz = cz;

			if (ny < 0 or n <= ny or nx < 0 or m <= nx or vis[ny][nx][nz])
			{
				continue;
			}

			if ('1' == board[ny][nx])
			{
				if (0 == nz)
				{
					continue;
				}
				--nz;
			}

			q.push({ ny, nx, nz });
			vis[ny][nx][nz] = vis[cy][cx][cz] + 1;
		}
	}

	cout << -1;

	return 0;
}
```
